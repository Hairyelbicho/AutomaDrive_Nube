    <script>
        const $ = (id) => document.getElementById(id);

        // --- RELOJ ---
        function startClock() {
            setInterval(() => {
                const d = new Date();
                $('reloj-maestro').textContent = d.toLocaleTimeString('es-ES', {hour12:false});
            }, 1000);
        }

        // --- RADIO Y AUDIO ---
        const audio = $('audio-stream');
        function changeStation() {
            const val = $('radio-selector').value;
            if(val === 'spotify') { window.open('https://open.spotify.com', '_blank'); return; }
            if(!val) { stopRadio(); return; }
            
            const name = $('radio-selector').options[$('radio-selector').selectedIndex].text;
            $('radio-station-name').textContent = name;
            audio.src = val;
            playRadio();
        }

        function playRadio() {
            audio.play();
            $('play-pause').innerHTML = '<i class="fas fa-pause"></i>';
            $('wave-container').style.display = 'flex';
        }

        function stopRadio() {
            audio.pause();
            $('play-pause').innerHTML = '<i class="fas fa-play"></i>';
            $('wave-container').style.display = 'none';
        }

        function toggleRadio() {
            if(audio.paused) playRadio(); else stopRadio();
        }

        // --- GRABACIÓN DE VOZ (Caja Negra) ---
        let mediaRecorder;
        let recording = false;

        async function toggleRecord() {
            if(!recording) {
                try {
                    const stream = await navigator.mediaDevices.getUserMedia({ audio: true });
                    mediaRecorder = new MediaRecorder(stream);
                    mediaRecorder.start();
                    recording = true;
                    $('rec-btn').classList.add('rec-active');
                    $('rec-status').innerHTML = '<span style="color:var(--danger)">● GRABANDO SEGURIDAD</span>';
                } catch(e) { alert("Micrófono no detectado"); }
            } else {
                mediaRecorder.stop();
                recording = false;
                $('rec-btn').classList.remove('rec-active');
                $('rec-status').innerHTML = "Grabación guardada";
                setTimeout(()=>{$('rec-status').innerHTML=""}, 3000);
            }
        }

        // --- DETECCIÓN ALTAVOCES (Simulada por WiFi) ---
        function checkSpeakers() {
            setTimeout(() => {
                $('wifi-status').innerHTML = '<i class="fas fa-wifi" style="color:var(--primary)"></i> Altavoces: Sony-Workshop';
            }, 3000);
        }

        // --- CONFIG ---
        function abrirConfig() { $('modalConfig').style.display = 'block'; }
        function cerrarConfig() { $('modalConfig').style.display = 'none'; }

        window.onload = () => {
            startClock();
            checkSpeakers();
        };

        function buscarDesdeWidget() { alert('Buscando: ' + $('widget-search').value); }
        function abrirTelegram() { window.open('https://web.telegram.org', '_blank'); }
    </script>
