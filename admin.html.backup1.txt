    <script>
        async function cargarLogs() {
            const res = await fetch('/logs');
            const data = await res.json();
            const contenedor = document.getElementById('contenedor-logs');
            contenedor.innerHTML = '';
            data.logs.forEach(log => {
                const div = document.createElement('div');
                div.className = 'log-item';
                div.onclick = () => cargarFicha(log[1]);
                div.innerHTML = `<strong>${log[1]}</strong><br><span>${log[3] || 'Registrar...'}</span><br><small style="opacity:0.5">${log[0]}</small>`;
                contenedor.appendChild(div);
            });
        }

        function ampliarMapa() {
            window.open("http://googleusercontent.com/maps.google.com/3", "_blank");
        }

        async function cargarFicha(m) {
            const res = await fetch(`/datos_cliente/${m}`);
            const data = await res.json();
            document.getElementById('matricula').value = m;
            document.getElementById('propietario').value = data.propietario || '';
            document.getElementById('modelo').value = data.modelo || '';
            document.getElementById('telefono').value = data.telefono || '';
            document.getElementById('km').value = data.km || '';
            document.getElementById('notas').value = data.notas || '';
            document.getElementById('miModal').style.display = "block";
        }

        async function guardarFicha() {
            const payload = {
                matricula: document.getElementById('matricula').value,
                propietario: document.getElementById('propietario').value,
                modelo: document.getElementById('modelo').value,
                telefono: document.getElementById('telefono').value,
                km: document.getElementById('km').value,
                notas: document.getElementById('notas').value,
                proxima_itv: ""
            };
            await fetch('/guardar_cliente', { method: 'POST', headers: {'Content-Type': 'application/json'}, body: JSON.stringify(payload) });
            alert("Guardado");
            document.getElementById('miModal').style.display = "none";
            cargarLogs();
        }

        function buscarTiendas() { alert("Buscando proveedores 24h en tu zona..."); }
        function cerrarModal() { document.getElementById('miModal').style.display = "none"; }

        setInterval(cargarLogs, 10000);
        cargarLogs();
