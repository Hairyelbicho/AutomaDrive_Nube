    <script>
        setInterval(() => { document.getElementById('reloj-maestro').innerText = new Date().toLocaleTimeString(); }, 1000);

        function nuevaFicha() { document.getElementById('miModal').style.display = "block"; }
        function cerrarModal() { document.getElementById('miModal').style.display = "none"; }
        function toggleActividad() { alert("Cargando actividad técnica..."); }

        // ESTA ES LA PARTE QUE ARREGLA EL ERROR DE COMUNICACIÓN
        async function guardarFicha() {
            const btn = document.getElementById('btn-guardar');
            const data = {
                matricula: document.getElementById('matricula').value.trim(),
                cliente: document.getElementById('propietario').value.trim(),
                whatsapp: document.getElementById('telefono').value.trim(),
                notas: document.getElementById('notes').value.trim()
            };

            if(!data.matricula) return alert("Matrícula obligatoria");

            btn.disabled = true;
            btn.innerHTML = '<i class="fas fa-spinner fa-spin"></i> ENVIANDO...';

            try {
                const response = await fetch('/guardar_cliente', {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify(data)
                });

                if (response.ok) {
                    alert("✅ Guardado correctamente en la Raspberry y la Nube.");
                    cerrarModal();
                    location.reload(); 
                } else {
                    alert("❌ Error interno del servidor. Revisa app.py.");
                }
            } catch (e) {
                // Aquí capturamos el error que te salía antes
                alert("❌ Fallo de comunicación: Verifica que la Raspberry Pi esté encendida y app.py esté corriendo.");
            } finally {
                btn.disabled = false;
                btn.innerHTML = '<i class="fas fa-save"></i> GUARDAR Y AVISAR';
            }
        }

        function buscarDesdeWidget() {
            const m = document.getElementById('widget-search').value.toUpperCase().trim();
            if(m) { document.getElementById('matricula').value = m; nuevaFicha(); }
        }
    </script>
